  var jeune = []; //creer tableau des jeunes
  var moyen = []; //creer tableau des moyens
  var vieux = []; //creer tableau des vieux
  while (true) {
      var age = prompt('Veuillez indiquer l\'age de la personne : \n\n + de 100 s\'arrete');
      if ((age <= 100) && (age > 40)) { // age entre 40 et 100ans inclus
          vieux.push(age);    //on ajoute l'age au tableau des vieux        
          age++;   //on incremente
      }
      else if ((age < 40) && (age >= 20)) { //age entre 20 inclus et 40 ans
          moyen.push(age);    //on ajoute l'age au tableau des moyens                  
          age++;   //on incremente
      }
      else if (age < 20) {  // age inferieur à 20 ans
          jeune.push(age);  //on ajoute l'age au tableau des jeunes
          age++;   //on incremente
      }
      else if ((age > 100) && (age != '')) {
          break;
      }
      }
      console.log('Voici l\'age des jeunes :'); 
      for (var age in jeune) {
          console.log(jeune[age] + 'ans');     //on affiche le tableau jeune
      }
      console.log('Voici l\'age des moyens :');
      for (var age in moyen) {                
          console.log(moyen[age] + 'ans ');   //on affiche le tableau moyens
      }
      console.log('Voici l\'age des vieux :');
      for (var age in vieux) {
          console.log(vieux[age] + 'ans ');   //on affiche le tableau vieux
      }
