          function afficheTab() {
              for (var i = 0, c = tab.length; i < c; i++) { //affiche element du tableau --> c -> compteur
                  console.table(tab[i]);
              }
          }
          let search = prompt("Rechercher un prenom :");
          let position = tab.indexOf(search);
          tab.splice(position, 1); // supprime 1 élément à la position du tableau
          tab.push(" (vide) ");
          console.log(search + ' a été supprimé et le tableau contient desormais: ');
          afficheTab();
